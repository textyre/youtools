import fs from 'fs'
import path from 'path'
import os from 'os'

/* ── Types ─────────────────────────────────────────── */

export interface SubtitlesConfig {
  outputDir: string
  defaultFormat: string
  defaultLanguage: string
  preferAutoGenerated: boolean
}

export interface CacheConfig {
  ttl: number
  dir: string
}

export interface AppConfig {
  subtitles: SubtitlesConfig
  cache: CacheConfig
  authorBonus: number
}

/* ── Defaults ──────────────────────────────────────── */

export const DEFAULTS: AppConfig = {
  subtitles: {
    outputDir: '~/youtools/subtitles',
    defaultFormat: 'srt',
    defaultLanguage: 'en',
    preferAutoGenerated: false,
  },
  cache: {
    ttl: 3600,
    dir: '~/.cache/youtools',
  },
  authorBonus: 1000,
}

/* ── Helpers ───────────────────────────────────────── */

export function resolveHome(p: string): string {
  if (p.startsWith('~/')) return path.join(os.homedir(), p.slice(2))
  return p
}

export function deepMerge<T extends Record<string, unknown>>(base: T, override: Record<string, unknown>): T {
  const result = { ...base } as Record<string, unknown>
  for (const key of Object.keys(override)) {
    const baseVal = result[key]
    const overVal = override[key]
    if (
      baseVal && overVal &&
      typeof baseVal === 'object' && !Array.isArray(baseVal) &&
      typeof overVal === 'object' && !Array.isArray(overVal)
    ) {
      result[key] = deepMerge(baseVal as Record<string, unknown>, overVal as Record<string, unknown>)
    } else {
      result[key] = overVal
    }
  }
  return result as T
}

/* ── Config loader ─────────────────────────────────── */

const CONFIG_PATH = path.join(os.homedir(), '.config', 'youtools', 'youtools.json')

let _cached: AppConfig | null = null

export function loadConfig(): AppConfig {
  if (_cached) return _cached
  let fileConfig: Record<string, unknown> = {}
  try {
    const raw = fs.readFileSync(CONFIG_PATH, 'utf8')
    fileConfig = JSON.parse(raw) as Record<string, unknown>
  } catch {
    // file missing or invalid — use defaults
  }
  _cached = deepMerge(DEFAULTS, fileConfig)
  return _cached
}

/** Reset cached config — for testing only */
export function _resetConfig(): void {
  _cached = null
}

import { execFile } from 'child_process'
import { promisify } from 'util'

const execFileAsync = promisify(execFile)

/* ── Types ─────────────────────────────────────────── */

export interface SubtitleTrack {
  lang: string
  name: string
  isAutoGenerated: boolean
}

export interface DownloadSubOpts {
  lang: string
  format: string
  autoGenerated: boolean
  outputDir: string
}

/* ── Validation ────────────────────────────────────── */

const VIDEO_ID_RE = /^[A-Za-z0-9_-]{1,20}$/

export function validateVideoId(id: string): boolean {
  return VIDEO_ID_RE.test(id)
}

/* ── Argument builders ─────────────────────────────── */

function videoUrl(videoId: string): string {
  return `https://www.youtube.com/watch?v=${videoId}`
}

export function buildSubArgs(videoId: string, opts: DownloadSubOpts): string[] {
  const subFlag = opts.autoGenerated ? '--write-auto-sub' : '--write-sub'
  return [
    subFlag,
    '--sub-lang', opts.lang,
    '--sub-format', opts.format,
    '--skip-download',
    '-o', `${opts.outputDir}/%(id)s.%(sub_lang)s.%(ext)s`,
    videoUrl(videoId),
  ]
}

/* ── Parsers ───────────────────────────────────────── */

export function parseListSubsOutput(json: Record<string, unknown>): SubtitleTrack[] {
  const tracks: SubtitleTrack[] = []

  const subs = json.subtitles as Record<string, { ext: string; name: string }[]> | undefined
  if (subs) {
    for (const [lang, entries] of Object.entries(subs)) {
      const name = entries[0]?.name ?? lang
      tracks.push({ lang, name, isAutoGenerated: false })
    }
  }

  const auto = json.automatic_captions as Record<string, { ext: string; name: string }[]> | undefined
  if (auto) {
    for (const [lang, entries] of Object.entries(auto)) {
      const name = entries[0]?.name ?? lang
      tracks.push({ lang, name, isAutoGenerated: true })
    }
  }

  return tracks
}

/* ── yt-dlp execution ──────────────────────────────── */

const TIMEOUT_MS = 30_000

export async function checkYtdlp(): Promise<void> {
  try {
    await execFileAsync('yt-dlp', ['--version'], { timeout: 5000 })
  } catch {
    throw new Error(
      'yt-dlp is not installed or not in PATH.\n' +
      'Install: brew install yt-dlp  (macOS)\n' +
      '         pip install yt-dlp    (pip)\n' +
      '  See: https://github.com/yt-dlp/yt-dlp#installation'
    )
  }
}

export async function listSubs(videoId: string): Promise<SubtitleTrack[]> {
  if (!validateVideoId(videoId)) throw new Error(`Invalid video ID: ${videoId}`)
  const { stdout } = await execFileAsync(
    'yt-dlp',
    ['--dump-json', '--skip-download', videoUrl(videoId)],
    { timeout: TIMEOUT_MS }
  )
  const json = JSON.parse(stdout) as Record<string, unknown>
  return parseListSubsOutput(json)
}

export async function downloadSubs(
  videoId: string,
  opts: DownloadSubOpts
): Promise<string> {
  if (!validateVideoId(videoId)) throw new Error(`Invalid video ID: ${videoId}`)
  const args = buildSubArgs(videoId, opts)
  await execFileAsync('yt-dlp', args, { timeout: TIMEOUT_MS })
  // yt-dlp writes file to outputDir — return expected path
  return `${opts.outputDir}/${videoId}.${opts.lang}.${opts.format}`
}

export async function getVideoInfo(videoId: string): Promise<Record<string, unknown>> {
  if (!validateVideoId(videoId)) throw new Error(`Invalid video ID: ${videoId}`)
  const { stdout } = await execFileAsync(
    'yt-dlp',
    ['--dump-json', '--skip-download', videoUrl(videoId)],
    { timeout: TIMEOUT_MS }
  )
  return JSON.parse(stdout) as Record<string, unknown>
}
```

import fs from 'fs'
import path from 'path'
import { resolveHome, loadConfig } from '../../config'
import { checkYtdlp, listSubs, downloadSubs, getVideoInfo as ytdlpVideoInfo } from '../../lib/ytdlp'
import type { SubtitlesCfg, SearchMatch } from './types'
import type { SubtitleTrack } from '../../lib/ytdlp'

/* ── SRT / VTT parsing ─────────────────────────────── */

export interface TimestampedEntry {
  timestamp: string
  text: string
}

export function parseSrt(content: string): TimestampedEntry[] {
  const entries: TimestampedEntry[] = []
  const blocks = content.trim().split(/\n\n+/)
  for (const block of blocks) {
    const lines = block.split('\n')
    if (lines.length < 3) continue
    // line 0: sequence number, line 1: timestamps, line 2+: text
    const timeMatch = lines[1].match(/(\d{2}:\d{2}:\d{2})/)
    if (!timeMatch) continue
    const text = lines.slice(2).join(' ').trim()
    if (text) entries.push({ timestamp: timeMatch[1], text })
  }
  return entries
}

export function searchInSubtitles(entries: TimestampedEntry[], query: string): SearchMatch[] {
  const q = query.toLowerCase()
  return entries
    .filter((e) => e.text.toLowerCase().includes(q))
    .map((e) => ({ timestamp: e.timestamp, text: e.text }))
}

/* ── Subtitle file helpers ─────────────────────────── */

function subtitlePath(outputDir: string, videoId: string, lang: string, format: string): string {
  return path.join(resolveHome(outputDir), `${videoId}.${lang}.${format}`)
}

function ensureDir(dir: string): void {
  fs.mkdirSync(resolveHome(dir), { recursive: true, mode: 0o755 })
}

/* ── Public API ────────────────────────────────────── */

export async function listLanguages(videoId: string): Promise<SubtitleTrack[]> {
  await checkYtdlp()
  return listSubs(videoId)
}

export async function fetchSubtitles(cfg: SubtitlesCfg): Promise<string> {
  await checkYtdlp()
  const outputDir = resolveHome(cfg.outputDir)
  const filePath = subtitlePath(cfg.outputDir, cfg.videoId, cfg.lang, cfg.format)

  // Check cache
  if (fs.existsSync(filePath)) {
    return fs.readFileSync(filePath, 'utf8')
  }

  // Download
  ensureDir(cfg.outputDir)
  await downloadSubs(cfg.videoId, {
    lang: cfg.lang,
    format: cfg.format,
    autoGenerated: cfg.autoGenerated,
    outputDir,
  })

  if (!fs.existsSync(filePath)) {
    throw new Error(
      `Subtitles not found after download. The video may not have ${cfg.lang} subtitles.\n` +
      `Use --list to see available languages.`
    )
  }

  return fs.readFileSync(filePath, 'utf8')
}

export async function fetchVideoInfo(videoId: string): Promise<{
  title: string
  channel: string
  duration: number
  description: string
  uploadDate: string
  viewCount: number
}> {
  await checkYtdlp()
  const info = await ytdlpVideoInfo(videoId)
  return {
    title: (info.title as string) ?? '',
    channel: (info.channel as string) ?? (info.uploader as string) ?? '',
    duration: (info.duration as number) ?? 0,
    description: (info.description as string) ?? '',
    uploadDate: (info.upload_date as string) ?? '',
    viewCount: (info.view_count as number) ?? 0,
  }
}

          left: '|', 'left-mid': '+', mid: '-', 'mid-mid': '+',
          right: '|', 'right-mid': '+', middle: '|',
        }
      : undefined,
  })

  for (const t of tracks) {
    table.push([t.lang, t.name, t.isAutoGenerated ? 'yes' : 'no'])
  }

  // eslint-disable-next-line no-console
  console.log(table.toString())
}

export function renderSearchResults(matches: SearchMatch[], ascii: boolean): void {
  if (matches.length === 0) {
    // eslint-disable-next-line no-console

  }

  const table = new Table({

    colWidths: [14, Math.max(60, (process.stdout.columns || 120) - 18)],
    wordWrap: true,
    style: { border: [], head: [] },

          top: '-', 'top-mid': '+', 'top-left': '+', 'top-right': '+',
          bottom: '-', 'bottom-mid': '+', 'bottom-left': '+', 'bottom-right': '+',
          left: '|', 'left-mid': '+', mid: '-', 'mid-mid': '+',

  })

  for (const m of matches) {

  }

  // eslint-disable-next-line no-console

  .option('--stdout', 'print subtitles to stdout instead of saving', false)
  .option('--ascii', 'force ASCII table', false)
  .action(async (videoId: string, opts: SubtitlesOpts) => {
    const config = loadConfig()

    // --list mode
    if (opts.list) {
      const tracks = await listLanguages(videoId)
      renderLanguagesTable(tracks, opts.ascii)
      return
    }

    const cfg: SubtitlesCfg = {
      videoId,
      lang: opts.lang ?? config.subtitles.defaultLanguage,
      format: (opts.format ?? config.subtitles.defaultFormat) as SubtitleFormat,
      autoGenerated: opts.auto || config.subtitles.preferAutoGenerated,
      outputDir: opts.output ?? config.subtitles.outputDir,
      stdout: opts.stdout,
      list: false,
      search: opts.search ?? null,
    }

    // Validate format
    if (!SUBTITLE_FORMATS.includes(cfg.format)) {
      throw new Error(`Invalid format "${cfg.format}". Allowed: ${SUBTITLE_FORMATS.join(', ')}`)
    }

    const content = await fetchSubtitles(cfg)

    // --search mode
    if (cfg.search) {
      const entries = parseSrt(content)
      const matches = searchInSubtitles(entries, cfg.search)
      renderSearchResults(matches, opts.ascii)
      return
    }

    // --stdout mode
    if (cfg.stdout) {
      // eslint-disable-next-line no-console
      console.log(content)
      return
    }

    const outPath = `${resolveHome(cfg.outputDir)}/${cfg.videoId}.${cfg.lang}.${cfg.format}`
    // eslint-disable-next-line no-console
    console.log(`Subtitles saved to: ${outPath}`)
  })
```
